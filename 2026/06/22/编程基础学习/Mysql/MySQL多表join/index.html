MySQL 多表查询示例

学生管理系统双表查询示例集合

示例1：查询学生及其选课信息

SELECT 
    s.student_id,
    s.name,
    s.major,
    e.course_id,
    e.status
FROM students s
JOIN enrollments e ON s.student_id = e.student_id
ORDER BY s.name, e.course_id
LIMIT 20;

示例2：查询学生选课的课程名称

SELECT 
    s.student_id,
    s.name,
    c.course_name,
    e.status
FROM students s
JOIN enrollments e ON s.student_id = e.student_id
JOIN courses c ON e.course_id = c.course_id
ORDER BY s.name
LIMIT 20;

示例3：查询选择了具体课程的学生信息

SELECT 
    c.course_name,
    s.student_id,
    s.name,
    s.major
FROM courses c
JOIN enrollments e ON c.course_id = e.course_id
JOIN students s ON e.student_id = s.student_id
WHERE c.course_name = '计算机科学导论'
ORDER BY s.name;

示例4：查询学生登录记录

SELECT 
    s.student_id,
    s.name,
    ll.login_datetime,
    ll.device_type
FROM students s
JOIN login_logs ll ON s.student_id = ll.student_id
ORDER BY ll.login_datetime DESC
LIMIT 20;

示例5：查询有成绩的学生选课信息

SELECT 
    s.name,
    c.course_name,
    e.grade,
    e.status
FROM students s
JOIN enrollments e ON s.student_id = e.student_id
JOIN courses c ON e.course_id = c.course_id
WHERE e.grade IS NOT NULL
ORDER BY e.grade DESC
LIMIT 15;

示例6：查询课程的选课学生数量

SELECT 
    c.course_name,
    c.course_code,
    COUNT(e.student_id) AS student_count
FROM courses c
LEFT JOIN enrollments e ON c.course_id = e.course_id
GROUP BY c.course_id, c.course_name, c.course_code
ORDER BY student_count DESC;

示例7：查询每个学生的选课数量

SELECT 
    s.student_id,
    s.name,
    s.major,
    COUNT(e.course_id) AS course_count
FROM students s
LEFT JOIN enrollments e ON s.student_id = e.student_id
GROUP BY s.student_id, s.name, s.major
ORDER BY course_count DESC;

示例8：查询课程安排及对应课程信息

SELECT 
    c.course_name,
    cs.classroom,
    cs.start_time,
    cs.end_time,
    cs.teacher_name
FROM courses c
JOIN course_schedule cs ON c.course_id = cs.course_id
ORDER BY cs.start_time;

示例9：查询计算机专业学生的选课情况

SELECT 
    s.name,
    s.major,
    c.course_name,
    e.status
FROM students s
JOIN enrollments e ON s.student_id = e.student_id
JOIN courses c ON e.course_id = c.course_id
WHERE s.major LIKE '%计算机%'
ORDER BY s.name;

示例10：查询已完成课程的学生成绩

SELECT 
    s.name,
    c.course_name,
    e.grade,
    e.status
FROM students s
JOIN enrollments e ON s.student_id = e.student_id
JOIN courses c ON e.course_id = c.course_id
WHERE e.status = '已完成'
ORDER BY e.grade DESC;
