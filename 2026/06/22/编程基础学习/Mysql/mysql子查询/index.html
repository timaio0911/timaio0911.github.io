MySQL 子查询实战案例

1. 标量子查询

案例1：查询高于平均年龄的学生信息

[!info]
业务场景：学生管理部门需要识别年龄偏大的学生，进行针对性管理
实际需求：找出年龄高于全校平均年龄的学生，用于学业指导

-- 标量子查询：计算全校平均年龄并筛选学生
SELECT 
    student_id AS 学生学号,
    name AS 学生姓名,
    YEAR(CURDATE()) - YEAR(birth_date) AS 学生年龄,
    major AS 专业
FROM students
WHERE YEAR(CURDATE()) - YEAR(birth_date) > (
    SELECT AVG(YEAR(CURDATE()) - YEAR(birth_date)) 
    FROM students 
    WHERE birth_date IS NOT NULL
)
AND birth_date IS NOT NULL
ORDER BY 学生年龄 DESC;

案例2：查询学分高于平均值的重点课程

[!info]
业务场景：教务管理需要识别高学分课程，合理安排教学资源
实际需求：找出学分超过平均学分的课程，重点配置师资

-- 标量子查询：筛选高学分重点课程
SELECT 
    course_id AS 课程编号,
    course_name AS 课程名称,
    credits AS 课程学分,
    department AS 开课院系
FROM courses
WHERE credits > (
    SELECT AVG(credits) 
    FROM courses 
    WHERE credits IS NOT NULL
)
ORDER BY credits DESC;

2. 行子查询

案例9：查找与指定学生年级和专业完全相同的同学

[!info]
业务场景：学生事务需要为特定学生找到同年级同专业的伙伴
实际需求：为学号1001的学生找到同年级同专业的其他学生

-- 行子查询：精确匹配多个属性
SELECT 
    student_id AS 学生学号,
    name AS 学生姓名,
    grade AS 年级,
    major AS 专业
FROM students
WHERE (grade, major) = (
    SELECT grade, major 
    FROM students 
    WHERE student_id = 1001
)
AND student_id != 1001
ORDER BY name;

案例10：查找生日月份和性别都相同的学生

[!info]
业务场景：学生活动组织需要按特征组织主题活动
实际需求：找出与特定学生生日月份和性别都相同的其他学生

-- 行子查询：多属性特征匹配
SELECT 
    student_id AS 学生学号,
    name AS 学生姓名,
    MONTH(birth_date) AS 出生月份,
    gender AS 性别
FROM students
WHERE (MONTH(birth_date), gender) = (
    SELECT MONTH(birth_date), gender
    FROM students
    WHERE student_id = 1002
)
AND student_id != 1002
AND birth_date IS NOT NULL
AND gender IS NOT NULL
ORDER BY name;

3. 列子查询

案例13：查询有登录记录的活跃学生信息

[!info]
业务场景：学生管理需要识别系统活跃用户，了解学生使用情况
实际需求：找出所有有系统登录记录的学生基本信息

-- 列子查询：识别活跃用户
SELECT 
    student_id AS 学生学号,
    name AS 学生姓名,
    major AS 专业,
    enrollment_date AS 入学日期
FROM students
WHERE student_id IN (
    SELECT DISTINCT student_id 
    FROM login_logs 
    WHERE student_id IS NOT NULL
)
ORDER BY enrollment_date DESC;

案例14：查询没有登录记录的沉默用户

[!info]
业务场景：系统推广需要识别未使用系统的学生进行引导
实际需求：找出从未登录过系统的学生名单

-- 列子查询：识别沉默用户
SELECT 
    student_id AS 学生学号,
    name AS 学生姓名,
    major AS 专业,
    enrollment_date AS 入学日期
FROM students
WHERE student_id NOT IN (
    SELECT DISTINCT student_id 
    FROM login_logs 
    WHERE student_id IS NOT NULL
)
ORDER BY enrollment_date;

4. 表子查询

案例19：专业学生数量统计分析

[!info]
业务场景：招生规划需要了解各专业学生分布情况
实际需求：统计学生人数≥3的专业详细信息

-- 表子查询：专业规模统计分析
SELECT 
    专业名称,
    学生人数,
    平均年龄
FROM (
    SELECT 
        major AS 专业名称,
        COUNT(*) AS 学生人数,
        AVG(YEAR(CURDATE()) - YEAR(birth_date)) AS 平均年龄
    FROM students
    WHERE major IS NOT NULL AND birth_date IS NOT NULL
    GROUP BY major
) AS 专业统计
WHERE 学生人数 >= 3
ORDER BY 学生人数 DESC;

案例20：院系课程配置情况分析

[!info]
业务场景：教务管理需要评估各院系课程配置的合理性
实际需求：分析各院系的课程数量和学分配置情况

-- 表子查询：院系课程配置分析
SELECT 
    院系名称,
    课程总数,
    总学分,
    平均学分
FROM (
    SELECT 
        department AS 院系名称,
        COUNT(*) AS 课程总数,
        SUM(credits) AS 总学分,
        AVG(credits) AS 平均学分
    FROM courses
    WHERE department IS NOT NULL AND credits IS NOT NULL
    GROUP BY department
) AS 院系统计
WHERE 课程总数 >= 2
ORDER BY 总学分 DESC;

二次分析的表子查询案例

案例1：年级分布分析

-- 统计各年级类型的数量
SELECT 
    年级类型,
    COUNT(*) AS 年级数量
FROM (
    SELECT 
        grade,
        CASE 
            WHEN COUNT(*) >= 20 THEN '大年级'
            WHEN COUNT(*) >= 10 THEN '中年级'
            ELSE '小年级'
        END AS 年级类型
    FROM students
    WHERE grade IS NOT NULL
    GROUP BY grade
) AS 年级分类
GROUP BY 年级类型;

案例2：院系规模分析

-- 统计不同规模院系的数量
SELECT 
    规模等级,
    COUNT(*) AS 院系数量,
    AVG(课程数) AS 平均课程数
FROM (
    SELECT 
        department,
        COUNT(*) AS 课程数,
        CASE WHEN COUNT(*) >= 5 THEN '大院系' ELSE '小院系' END AS 规模等级
    FROM courses
    WHERE department IS NOT NULL
    GROUP BY department
) AS 院系规模
GROUP BY 规模等级;
