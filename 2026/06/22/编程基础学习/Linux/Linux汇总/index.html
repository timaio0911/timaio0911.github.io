Part3 – 常用命令

3.1 移动工作区域 cd

cd        # 移动
cd /      # 移动到根目录
cd ~      # 移动到家目录
cd -      # 回到移动前目录
.         # 当前目录
..        # 上级目录

3.2 查看

查看文件夹 ls

ls        # 只查看文件名
ls -l     # 查看所有信息（同 ll）
ls -h     # 查看人类可读大小
ls -a     # 查看所有文件包括隐藏文件

查看文件 cat

cat           # 查看文件全部显示
cat -n        # 显示行号
cat >         # 新建并写入（Ctrl+D 结束并保存）
cat >>        # 追加内容到文件末尾
less test.txt # 分页查看
  /           # 关键词查询
  q           # 退出
  PageUp/Down # 翻页
tail          # 只看文件末尾更新内容
tail -f       # 实时查看更新
more test.txt # 分页查看（只能往下看）
cat a.txt > b.txt  # 复制文件内容

touch             # 新建文件

# 剪切
mv 路径_文件/目录名 新路径_文件/目录名
# 示例：在根目录将 b3 剪切到 b1
mv /home/b1/b2/b3 /home/b1

# 复制（目录需要 -r 递归）
cp 路径_文件/目录名 新路径_文件/目录名
# 示例：在根目录将 b3 复制到 b1
cp -r home/b1/b2/b3 home/b1

rm              # 删除文件
rm -f           # 强制删除
rm -rf          # 强制删除文件夹
rm -f *.log     # 删除所有 .log 结尾文件
rm -f test*     # 删除所有 test 开头的文件
rm -rf 目录/*   # 删除目录内所有文件（清空弹夹）

3.6 搜索

找行 grep

grep 关键词 文件          # 文件中寻找关键词
grep -c 关键词 文件       # 只统计匹配到多少行
grep -v 关键词 文件       # 反向匹配
grep -n 关键词 文件       # 显示行号
grep -i 关键词 文件       # 忽略大小写
grep -rn 关键词 文件      # 递归搜索整个文件夹中所有文件

找列 cut

cut -d分隔符 -f第几列 文件名
# -f1      第一列
# -f1,3    第 1、3 列
# -f1-5    第 1 到 5 列

3.7 排序 sort

sort          # 基础排序
sort -n       # 按数值大小排序（小到大）
sort -nr      # 降序反转
sort -u       # 去重排列
sort -t       # 指定分隔符
sort -kX      # 指定排序第 X 列
  sort -k2,2  # 精确按照第 2 列排序
sort -n 文件名 -o 文件名  # 排序后覆盖写入原文件

组合使用

sort -t, -k5,5n    # 用逗号分隔，第五列数字升序
sort -t: -k3,3nr   # 用冒号分隔，第三列数值降序

dnf install 安装包 -y    # 下载
dnf remove 安装包 -y     # 删除

du        # 显示当前目录下所有子目录大小

du -s     # 显示总大小

find [路径] [表达式选项]

# 示例：查找所有 .csv 后缀的数据文件
find /home/user/credit_system -name "*.csv"

# 示例：查找所有大于 100K 的备份文件
find /home/user/credit_system/data/archive -name "*.bak.*" -size +100k

3.12 改变权限 chmod

[!info]
权限分为读(r=4), 写(w=2), 执行(x=1)
三个数字的和可以分别表示用户、组、其他这三个类别分别的权限
例如 7=4+2+1 表示三个权限都有

ls -l products.csv        # 查看当前权限，应为 -rw-r--r--
chmod u-w products.csv    # u=user, -w=移除写权限
ls -l products.csv        # 查看修改后权限，应为 -r--r--r--
chmod u+w products.csv    # 恢复写权限

Part5 – 三大剑客

1. awk

基础语法

awk '[/pattern/] {action}' 文件名

工作原理：

  1. awk 逐行扫描文件
  2. 对于每一行，检查是否匹配 pattern
  3. 如果匹配，就执行 action
  4. 如果省略 pattern，则对所有行执行 action

内建变量：

  • $0：整行内容
  • $1, $2, ...：第1个字段，第2个字段…
  • NF：当前行的字段总数
  • NR：当前行的行号
  • -F '分隔符'：指定字段分隔符

场景示例

场景1：格式化输出

awk -F',' '{print "申请ID:", $1, " | 客户:", $3, " | 申请金额:", $6, "元"}' applications.csv

场景2：条件过滤

# 查询所有申请金额超过 20 万的申请
awk -F',' 'NR > 1 && $6 > 200000 {print $0}' applications.csv

场景3：统计计算

awk -F',' 'BEGIN{count=0;total=0;}
    $7 == "通过" {
        count++;
        total += $6;
    }
    END {
        if (count > 0) {
            avg = total / count;
            print "通过申请总数:", count;
            print "通过申请总金额:", total, "元";
            print "通过申请平均金额:", avg, "元";
        } else {
            print "没有通过的申请记录。";
        }
    }' applications.csv

2. sed

# 查找输出（只打印匹配的行）
sed -n '/2025-06/p' applications.csv

# 将文件中男换成女（不改变原文件）
sed 's/男/女/g' user.csv

# 替换完后写入原文件
sed -i 's/旧内容/新内容/g' file.txt

# 替换后自动生成 .bak 备份文件
sed -i.bak 's/旧内容/新内容/g' file.txt

# 删除匹配行
sed '/2025-01/d' data.csv

# 删除第一行
sed '1d' file.csv

3. 计算

[!tip]
用 (()) 来进行计算

#!/bin/bash

num1=10
num2=3

# 使用 $(()) 进行算术运算
sum=$((num1 + num2))
diff=$((num1 - num2))
product=$((num1 * num2))
quotient=$((num1 / num2))
remainder=$((num1 % num2))

echo "$num1 + $num2 = $sum"
echo "$num1 - $num2 = $diff"
echo "$num1 * $num2 = $product"
echo "$num1 / $num2 = $quotient"
echo "$num1 % $num2 = $remainder"

# 使用 let 命令
let result=num1*num2+5
echo "复合运算结果：$result"

#!/bin/bash

# 列表式 for 循环
echo "=== 处理文件列表 ==="
for file in *.txt *.log; do
    if [ -f "$file" ]; then
        echo "处理文件：$file"
        wc -l "$file"
    fi
done

echo "=== 年份统计 ==="

    echo "年份：$year"
done

while 循环

#!/bin/bash

counter=1
while [ $counter -le 5 ]; do
    echo "while循环：第 $counter 次"
    ((counter++))
done

6. 多分支判断 case

#!/bin/bash

echo "请选择操作："
echo "1) 查看文件"
echo "2) 创建文件"
echo "3) 删除文件"
echo "4) 退出"
read -p "请输入选择 (1-4)：" choice

case $choice in
    1)
        echo "您选择了查看文件"
        ls -la
        ;;
    2)
        echo "您选择了创建文件"
        read -p "请输入文件名：" filename
        touch "$filename"
        echo "文件 $filename 已创建"
        ;;
    3)
        echo "您选择了删除文件"
        read -p "请输入要删除的文件名：" filename
        if [ -f "$filename" ]; then
            rm -f "$filename"
            echo "文件 $filename 已删除"
        else
            echo "文件不存在"
        fi
        ;;
    4)
        echo "再见！"
        exit 0
        ;;
    *)
        echo "无效选择，请输入1-4"
        ;;
esac

7. 数组

#!/bin/bash

# 数组定义
servers=("web1" "web2" "db1" "cache1")
ports=(80 8080 3306 6379)

# 数组访问
echo "第一个服务器：${servers[0]}"
echo "所有服务器：${servers[@]}"
echo "数组长度：${#servers[@]}"

# 数组遍历
for i in "${!servers[@]}"; do
    echo "服务器 ${servers[i]} 端口 ${ports[i]}"
done
